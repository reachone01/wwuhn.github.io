&lt;p>增加对函数返回值的检验。&lt;/p>

//函数定义时的输入的合法性检查
//函数调用时输出的检验

#include <iostream>
using namespace std;
long fact(int n);

int main()
{
    int n;
    int ret;
    printf("Input n:");
    scanf("%d",&n);
    ret = fact(n);
    if(ret==-1)      // 函数返回值的检验 
        printf("Input data error!");
    else
        printf("%d! = %d\n",n,ret);

    while(1);
    return 0;
}

long fact(int n)
{
    int ret = 1;
    if(n<0)         // 在函数的入口处，检查输入参数的合法性 
    {
        //printf("Input data error!");
        return -1;
    }
    else
    {
        for(int i=2;i<=n;i++)
            ret *= i;
        return ret;
    }
}
